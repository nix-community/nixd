27 void consume(std::size_t N = 1) {
28 assert(Cur.Offset + N <= Src.length());
29 // Update Line & Column & Offset
30 for (std::size_t I = 0; I < N; ++I) {
31 if (Src[Cur.Offset + I] == '\n') {
32 ++Cur.Line;
33 Cur.Column = 0;
34 } else {
35 ++Cur.Column;
36 }
37 }
38 Cur.Offset += N;
39 }
